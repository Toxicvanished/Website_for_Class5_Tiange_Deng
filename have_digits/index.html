def have_digits(s: str) -> int:
    """
    Check if a string has digits in it.
    Input: a random string;
    Output: int, 0 for no digit detected, 1 for digit detected.
    """

    out = 0

    # loop through the string
    for c in s:
        # check if the character is a digit
        if c.isdigit():
            out = 1
            break

    return out
